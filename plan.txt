2️⃣ Use Row Level Security (RLS)

This is your backend security layer.

Example:

Employee can only see their own attendance:

CREATE POLICY "Employees can view own attendance"
ON attendance
FOR SELECT
USING (auth.uid() = user_id);

Admin can see all:

USING (EXISTS (
  SELECT 1 FROM users
  WHERE id = auth.uid()
  AND role = 'admin'
));

If you configure RLS correctly, you don’t need heavy backend.

🏗️ FINAL DATABASE STRUCTURE
1️⃣ Profiles (Users Extension)

Supabase manages:

auth.users

We create:

CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  phone TEXT,
  role TEXT CHECK (role IN ('admin','hr','emp')) NOT NULL DEFAULT 'emp',
  emp_id TEXT,
  designation TEXT,
  document_submit BOOLEAN DEFAULT FALSE,
  document_received BOOLEAN DEFAULT FALSE,
  total_leaves INT DEFAULT 12,
  remaining_leaves INT DEFAULT 12,
  add_on_leaves INT DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

Indexes:

CREATE INDEX idx_profiles_role ON profiles(role);
CREATE INDEX idx_profiles_email ON profiles(email);
2️⃣ Company Settings

Stores office rules.

CREATE TABLE company_settings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  office_lat DECIMAL,
  office_lng DECIMAL,
  allowed_radius_meters INT DEFAULT 100,
  allowed_ip_range TEXT,
  gps_login_required BOOLEAN DEFAULT FALSE,
  ip_login_required BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW()
);

Only one row needed.

3️⃣ Company Calendar (Holiday + Weekend Engine)
CREATE TABLE company_calendar (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  date DATE UNIQUE NOT NULL,
  type TEXT CHECK (type IN ('working','holiday','weekend')) NOT NULL,
  description TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

Index:

CREATE INDEX idx_calendar_date ON company_calendar(date);

Weekends auto-inserted via script.

4️⃣ Attendance

This is daily record per employee.

CREATE TABLE attendance (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  date DATE NOT NULL,
  present BOOLEAN DEFAULT FALSE,
  check_in TIMESTAMP,
  check_out TIMESTAMP,
  total_minutes INT DEFAULT 0,
  late_by INT DEFAULT 0,
  overtime_by INT DEFAULT 0,
  work_type TEXT CHECK (work_type IN ('office','wfh')),
  ip_address TEXT,
  latitude DECIMAL,
  longitude DECIMAL,
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(user_id, date)
);

Indexes:

CREATE INDEX idx_attendance_user_date ON attendance(user_id, date);
CREATE INDEX idx_attendance_date ON attendance(date);
5️⃣ Tasks
CREATE TABLE tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  assigned_to UUID REFERENCES profiles(id) ON DELETE CASCADE,
  start_day DATE NOT NULL,
  end_day DATE NOT NULL,
  status TEXT CHECK (status IN ('pending','in_progress','completed')) DEFAULT 'pending',
  priority TEXT CHECK (priority IN ('low','medium','high','critical')) DEFAULT 'medium',
  created_by UUID REFERENCES profiles(id),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

Indexes:

CREATE INDEX idx_tasks_assigned ON tasks(assigned_to);
CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_priority ON tasks(priority);
6️⃣ Leave Requests
CREATE TABLE leave_requests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  category TEXT CHECK (category IN ('leave','wfh','halfday')) NOT NULL,
  start_day DATE NOT NULL,
  end_day DATE NOT NULL,
  reason TEXT,
  approved BOOLEAN DEFAULT FALSE,
  approved_by UUID REFERENCES profiles(id),
  created_at TIMESTAMP DEFAULT NOW()
);

Indexes:

CREATE INDEX idx_leave_user ON leave_requests(user_id);
CREATE INDEX idx_leave_approved ON leave_requests(approved);
